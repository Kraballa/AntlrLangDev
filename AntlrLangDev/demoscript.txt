function roll(){
    return (rand() * 6)|int;
}

function pow(val, power){
    power = power|int;
    ret = 1;
    i = 0;
    while(i < power){
        ret = ret * val;
        i = i+1;
    }
    return ret;
}

print(pow(4, 1));
print(pow(4, 2));
print(pow(4, 3));
print(pow(4, 4));
